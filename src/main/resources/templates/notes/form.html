function addTag(tagName) { // This function is kept for compatibility with existing tag suggestions const tagInput = document.getElementById('tagNames'); const currentTags = tagInput.value.split(',').map(t => t.trim()).filter(t => t); if (!currentTags.includes(tagName)) { currentTags.push(tagName); tagInput.value = currentTags.join(', '); } } // Note linking functionality let selectedNoteLinks = []; function searchNotes() { const query = document.getElementById('noteSearch').value.trim(); if (query.length < 2) { document.getElementById('noteSearchResults').style.display = 'none'; return; } fetch(`/api/notes/search?query=${encodeURIComponent(query)}`) .then(response => response.json()) .then(notes => { displaySearchResults(notes); }) .catch(error => { console.error('Error searching notes:', error); }); } function displaySearchResults(notes) { const resultsContainer = document.getElementById('searchResultsList'); const resultsSection = document.getElementById('noteSearchResults'); if (notes.length === 0) { resultsContainer.innerHTML = '
No notes found
'; } else { resultsContainer.innerHTML = notes.map(note => `
${note.title}
${note.content ? note.content.substring(0, 60) + '...' : 'No content'}
`).join(''); } resultsSection.style.display = 'block'; } function selectNoteForLinking(noteId, noteTitle) { // Show modal to select relationship type and weight showLinkModal(noteId, noteTitle); } function showLinkModal(noteId, noteTitle) { const modal = document.createElement('div'); modal.className = 'modal fade'; modal.innerHTML = `
Link to: ${noteTitle}
Light (1) Weight: 5 Critical (10)
`; document.body.appendChild(modal); const bsModal = new bootstrap.Modal(modal); bsModal.show(); // Update weight display document.getElementById('linkWeight').addEventListener('input', function() { document.getElementById('weightDisplay').textContent = this.value; }); // Clean up modal after close modal.addEventListener('hidden.bs.modal', function() { document.body.removeChild(modal); }); } function addNoteLink(noteId, noteTitle) { const linkType = document.getElementById('linkType').value; const linkWeight = document.getElementById('linkWeight').value; // Check if already added if (selectedNoteLinks.find(link => link.noteId === noteId)) { alert('This note is already linked!'); return; } const link = { noteId: noteId, noteTitle: noteTitle, linkType: linkType, weight: parseInt(linkWeight) }; selectedNoteLinks.push(link); updateSelectedLinksDisplay(); updateHiddenInput(); // Close modal const modal = bootstrap.Modal.getInstance(document.querySelector('.modal')); modal.hide(); } function updateSelectedLinksDisplay() { const container = document.getElementById('selectedLinks'); if (selectedNoteLinks.length === 0) { container.innerHTML = `
Links you add will appear here. You can set relationship types and weights.
`; return; } container.innerHTML = selectedNoteLinks.map((link, index) => `
${link.noteTitle}
${link.linkType} Weight: ${link.weight}
`).join(''); } function removeNoteLink(index) { selectedNoteLinks.splice(index, 1); updateSelectedLinksDisplay(); updateHiddenInput(); } function updateHiddenInput() { document.getElementById('noteLinks').value = JSON.stringify(selectedNoteLinks); } // File queue management let fileQueue = []; // File attachment preview functionality with queue support function handleFileSelection() { const fileInput = document.getElementById('attachments'); // Add new files to queue for (let i = 0; i < fileInput.files.length; i++) { const file = fileInput.files[i]; // Check if file already exists in queue const exists = fileQueue.some(queuedFile => queuedFile.name === file.name && queuedFile.size === file.size ); if (!exists) { fileQueue.push(file); } } // Clear the input to allow selecting the same files again fileInput.value = ''; updateFilePreview(); updateFileInput(); } function updateFilePreview() { const preview = document.getElementById('filePreview'); const fileList = document.getElementById('fileList'); if (fileQueue.length === 0) { preview.style.display = 'none'; return; } let fileListHtml = ''; fileQueue.forEach((file, index) => { const fileSize = formatFileSize(file.size); const fileIcon = getFileIcon(file.type); fileListHtml += `
${file.name}
${fileSize}
`; }); fileList.innerHTML = fileListHtml; preview.style.display = 'block'; } function updateFileInput() { // Create a new DataTransfer object to update the file input const fileInput = document.getElementById('attachments'); const dt = new DataTransfer(); fileQueue.forEach(file => { dt.items.add(file); }); fileInput.files = dt.files; } function getFileIcon(mimeType) { if (mimeType.includes('word') || mimeType.includes('document')) { return 'fas fa-file-word text-primary'; } else if (mimeType.includes('pdf')) { return 'fas fa-file-pdf text-danger'; } else if (mimeType.includes('image')) { return 'fas fa-file-image text-success'; } else { return 'fas fa-file text-secondary'; } } function formatFileSize(bytes) { if (bytes < 1024) return bytes + ' B'; if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB'; if (bytes < 1024 * 1024 * 1024) return (bytes / 1024 / 1024).toFixed(1) + ' MB'; return (bytes / 1024 / 1024 / 1024).toFixed(1) + ' GB'; } function clearFileSelection() { fileQueue = []; document.getElementById('attachments').value = ''; document.getElementById('filePreview').style.display = 'none'; } function removeFileFromQueue(index) { fileQueue.splice(index, 1); updateFilePreview(); updateFileInput(); } // Auto-search on input document.addEventListener('DOMContentLoaded', function() { const searchInput = document.getElementById('noteSearch'); if (searchInput) { let searchTimeout; searchInput.addEventListener('input', function() { clearTimeout(searchTimeout); searchTimeout = setTimeout(() => { if (this.value.length >= 2) { searchNotes(); } else { document.getElementById('noteSearchResults').style.display = 'none'; } }, 300); }); } // File input change handler const fileInput = document.getElementById('attachments'); if (fileInput) { fileInput.addEventListener('change', handleFileSelection); } });
